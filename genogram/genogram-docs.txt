let gen = new Genogram();

gen.createNode(name, description)
create a new node with the given name and description
    name (string) - the name of the node
    description (string) - the description of the node
returns type GenogramNode

GenogramNode.addChild(childName)
add a child to the node
    childName (string/object) - string: the name of the child node, object: the child node

GenogramNode.addPartner(partnerName)
add a partner to the node
    partnerName (string/object) - string: the name of the partner node, object: the partner node

gen.getNode(name)
get a node by name
    name (string) - the name of the node

gen.getAllNodes()
get all nodes
